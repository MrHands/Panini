116 inline void Visit(Writer& writer) final
117 {
118 const BraceBreakingStyle breakingStyle =
119 m_options.breakingStyle == BraceBreakingStyle::Inherit
120 ? writer.GetBraceBreakingStyle()
121 : m_options.breakingStyle;
122
123 const bool wasNewLine = writer.IsOnNewLine();
124
125 switch (breakingStyle)
126 {
127
128 case BraceBreakingStyle::Attach:
129 {
130 writer << m_options.chunkBraceOpen << IndentPush() << NextLine();
131 m_callback(writer);
132 writer << IndentPop() << m_options.chunkBraceClose;
133
134 } break;
135
136 case BraceBreakingStyle::Allman:
137 {
138 if (!wasNewLine)
139 {
140 writer << NextLine();
141 }
142
143 writer << m_options.chunkBraceOpen << IndentPush() << NextLine();
144 m_callback(writer);
145 writer << IndentPop() << m_options.chunkBraceClose;
146
147 } break;
148
149 case BraceBreakingStyle::Whitesmiths:
150 {
151 if (!wasNewLine)
152 {
153 writer << NextLine() << IndentPush();
154 }
155
156 writer << m_options.chunkBraceOpen << NextLine();
157 m_callback(writer);
158 writer << m_options.chunkBraceClose;
159
160 if (!wasNewLine)
161 {
162 writer << IndentPop();
163 }
164
165 } break;
