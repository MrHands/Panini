144 inline virtual void Write(const std::string& chunk) override
145 {
146 if (!m_initialized)
147 {
148 return;
149 }
150
151 // line numbers
152
153 if (IsOnNewLine())
154 {
155 std::string padded = std::to_string(m_cursorY + 1);
156 for (size_t i = padded.length(); i < m_debugConfig.lineNumberPadding; ++i)
157 {
158 padded.insert(padded.begin(), ' ');
159 }
160
161 SetColor(Colors::White, Colors::Black);
162 WriteChunk(padded + " ");
163 ResetStyles();
164 }
165
166 const std::string& indentStr = GetConfig().chunkIndent;
167
168 // new line
169
170 if (chunk == GetConfig().chunkNewLine)
171 {
172 SetColor(Colors::Fuchsia, Colors::White);
173 WriteChunk("<LF>");
174 ResetStyles();
175
176 SetCursorPosition(0, m_cursorY + 1);
177
178 // handle input
179
180 if (m_isDebugging)
181 {
182 HandleInput("Press <Enter> to continue debugging, <Q> to quit");
183 }
184 }
185
186 // indentation
187
188 else if (
189 chunk.substr(0, indentStr.length()) == indentStr)
190 {
191 size_t offset = 0;
192 int32_t count = 0;
193
194 while (chunk.substr(offset, indentStr.length()) == indentStr)
195 {
196 SetColor(
197 (count % 2 == 0) ? Colors::Yellow : Colors::Cyan,
198 Colors::White
199 );
200 WriteChunk("-> ");
201
202 offset += indentStr.length();
203 count++;
204 }
205
206 ResetStyles();
207 }
208
209 // other chunks
210
211 else
212 {
213 WriteChunk(chunk);
214 }
215 }

290 inline void HandleInput(const std::string& message)
291 {
292 SetCursorPosition(0, m_cursorY + 1);
293 SetColor(Colors::White | Colors::Light, Colors::Black);
294 std::cout << message;
295 ResetStyles();
296
297 // wait for input
298
299 #ifdef _WINDOWS
300 HANDLE input = ::GetStdHandle(STD_INPUT_HANDLE);
301
302 while (1)
303 {
304 INPUT_RECORD record;
305 DWORD read = 0;
306 ::ReadConsoleInputA(input, &record, 1, &read);
307 if (record.EventType == KEY_EVENT)
308 {
309 KEY_EVENT_RECORD& keyEvent = record.Event.KeyEvent;
310
311 // wait for the key to be released
312
313 if (keyEvent.bKeyDown)
314 {
315 continue;
316 }
317
318 // stop debugging when Q is pressed
319
320 m_isDebugging = keyEvent.wVirtualKeyCode != 'Q';
321
322 break;
323 }
324 }
325 #endif
326
327 // clear line
328
329 #ifdef _WINDOWS
330 DWORD written = 0;
331 ::FillConsoleOutputCharacterA(
332 m_output,
333 ' ',
334 message.length(),
335 m_cursor,
336 &written
337 );
338
339 ::FillConsoleOutputAttribute(
340 m_output,
341 m_screenInfo.wAttributes,
342 message.length(),
343 m_cursor,
344 &written
345 );
346 #endif
347
348 // reset cursor
349
350 SetCursorPosition(0, m_cursorY - 1);
351 }
