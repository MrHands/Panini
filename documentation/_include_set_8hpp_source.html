162 inline void Sort(IncludeStyle resolvedStyle)
163 {
164 if (resolvedStyle == IncludeStyle::Inherit)
165 {
166 return;
167 }
168
169 // resolve priority for all entries
170
171 for (IncludeEntry& entry : m_entries)
172 {
173 IncludeStyle style =
174 (entry.style == IncludeStyle::Inherit)
175 ? resolvedStyle
176 : entry.style;
177
178 // base priority on include style
179
180 entry.priority = 1000;
181
182 switch (style)
183 {
184
185 case IncludeStyle::AngularBrackets:
186 entry.priority = 0;
187 break;
188
189 case IncludeStyle::DoubleQuotes:
190 entry.priority = 100;
191 break;
192
193 case IncludeStyle::SingleQuotes:
194 entry.priority = 200;
195 break;
196
197 default:
198 break;
199
200 }
201
202 // paths with folders should come before files
203
204 const std::string path = entry.path.string();
205
206 size_t offset = 0;
207 size_t next = 0;
208 while ((next = path.find_first_of('/', offset)) != std::string::npos)
209 {
210 entry.priority--;
211 offset = next + 1;
212 }
213 }
214
215 // sort by priority and path
216
217 std::sort(
218 m_entries.begin(),
219 m_entries.end(),
220 [](IncludeEntry& left, IncludeEntry& right) {
221 if (left.priority != right.priority)
222 {
223 return left.priority < right.priority;
224 }
225 else
226 {
227 return left.path < right.path;
228 }
229 }
230 );
231 }
