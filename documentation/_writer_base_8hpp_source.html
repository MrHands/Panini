121 {
122 if (m_state == State::NewLine)

168 inline WriterBase& operator << (const NextLine& command)
169 {
170 if (m_lineChunkCountWritten == 0 &&

173 // edge-case for empty lines within a comment block

194 inline WriterBase& operator << (const IndentPush& command)
195 {
